cmake_minimum_required(VERSION 3.15)
project(print-option-direct-test)

find_package(Python COMPONENTS Interpreter REQUIRED)

set(PYTHON_SCRIPT_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../../../src")
set(OUTPUT_FILE "${CMAKE_CURRENT_BINARY_DIR}/version.hpp")

add_test(
    NAME Print_Option_Direct_Check
    COMMAND ${Python_EXECUTABLE} -m version_builder 
            --lang cpp 
            --source git 
            --input ${CMAKE_CURRENT_SOURCE_DIR}/../../..
            --print
            ${OUTPUT_FILE}
)

set_property(TEST Print_Option_Direct_Check PROPERTY 
    PASS_REGULAR_EXPRESSION "This file is autogenerated."
)

set_property(TEST Print_Option_Direct_Check PROPERTY 
    ENVIRONMENT "PYTHONPATH=${PYTHON_SCRIPT_DIR}"
)
